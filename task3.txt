public class Main {

    private static void lala(int[] a){
        for (int i = 0; i < 12; i++) {
            if (a[i] < 6) {
                a[i] = a[i] * 2;
            }
        }
    }
    public static void main(String[] args) {
        int[] a = {1, 5, 3, 2, 11, 4, 5, 2, 4, 8, 9, 1};
        lala(a);
        for (int i = 0; i < 12; i++) {
            System.out.print(a[i]+" ");
        }
    }
}
