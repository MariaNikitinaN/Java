public class Main {
    
    public static void main(String[] args) {
        int[][] a = {{1,2},{6,3}};
        int i = 0;
        int j = 0;

        for (i = 1; i < a.length; i++) {
            for (j = 1; j < a.length; j++)
                a[i][i] = 1;
            for (i = 0; i < a.length; i++) {
                for (j = 0; j < a.length; j++)
                    System.out.printf("%3d", a[i][j]);
                System.out.println();
            }
        }
    }
}
